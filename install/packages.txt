# dotfiles-Fedora/install/packages.txt
# dnf package names for the modern terminal stack. One per line; # comments ok.
# Tools NOT reliably packaged on Fedora (starship, atuin, yazi) are installed
# by bootstrap.sh via their upstream installers instead — keep them OUT of here.

# ── shell + multiplexer ───────────────────────────────────────
zsh
tmux

# ── modern CLI core (all in Fedora repos as of F40+) ──────────
eza            # ls
bat            # cat
fd-find        # find  (binary: fd on Fedora, not fdfind)
ripgrep        # grep
zoxide         # cd
fzf            # fuzzy finder
git-delta      # git pager / diffs
btop           # top/htop
tealdeer       # tldr (fast Rust client; binary: tldr)

# ── modern CLI extras (core-doctor probes these; all in Fedora repos) ─
duf            # df   (disk usage/free)
procs          # ps
sd             # sed  (fast find/replace)
yq             # jq-for-YAML — Fedora's `yq` IS mikefarah's Go build (python-yq is the separate Python tool)
glow           # markdown viewer
gum            # shell-script UI toolkit (charm)
# dust, xh, doggo, carapace, sesh, gron, op (1Password CLI): NOT reliably in Fedora
# repos — bootstrap.sh installs them via cargo / go / the 1Password repo. gron was
# orphaned and dropped from Fedora (last shipped F41/42), so `dnf install gron` now
# fails — hence the go-install fallback.

# ── editor + git workflow ─────────────────────────────────────
neovim
git
git-subtree      # `git subtree` ships as a SEPARATE package on Fedora (architecture needs it)
openssh-clients  # ssh / ssh-keygen / ssh-agent — Workstation has it; minimal/Server may not
# lazygit is NOT in Fedora base repos — bootstrap.sh installs it via COPR (atim/lazygit)

# ── clipboard (Fedora Workstation is Wayland-first) ───────────
wl-clipboard   # wl-copy / wl-paste
xclip          # X11 fallback (sessions / SSH X-forward)

# ── build + language toolchain (needed for cargo-installed tools) ─
gcc
gcc-c++        # C++ compiler — Fedora's gcc is C-only; some tree-sitter parsers ship C++ scanners
make
cargo          # rust toolchain -> cargo install fallback for yazi/dust/xh

# ── Neovim provider deps (fix :checkhealth; use dnf NOT pip — F44 python is PEP668-managed)
python3-neovim # pynvim — the Python 3 provider module
python3-pip    # pip for the system python
tree-sitter-cli # nvim-treesitter (main branch) compiles parsers locally — needs the CLI ≥0.26.1.
                # If :checkhealth nvim-treesitter flags it as too old, drop it from this list and
                # install current via mise (`mise use -g tree-sitter`) or `cargo install tree-sitter-cli`.
# rust-analyzer  # uncomment if you use the Rust LSP (rustaceanvim wants it)
# golang         # uncomment if you want the Go LSP (gopls) + Go tooling

# ── misc useful ───────────────────────────────────────────────
curl
wget           # F42+ has no package literally named `wget`: it's a virtual capability
               # whose default provider is `wget2-wget` (wget2). `dnf install wget`
               # still resolves, but installs wget2 — pin `wget1-wget` if you need
               # classic GNU wget semantics (--show-progress, FTP, WARC).
jq
htop
flatpak
gawk
direnv
